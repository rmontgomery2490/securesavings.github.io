// Vehicle fields logic

let vehicleCount = 0;

document.getElementById('addVehicleBtn').addEventListener('click', function() {

if(vehicleCount >= 4) return;

vehicleCount++;

const vehicleDiv = document.createElement('div');

vehicleDiv.className = "row mb-2";

vehicleDiv.innerHTML = `

<div class="col-md-3 mb-2">

<input type="number" class="form-control" name="vehicleYear${vehicleCount}" placeholder="Year" min="1990" max="2025" required>

</div>

<div class="col-md-5 mb-2">

<input type="text" class="form-control" name="vehicleMake${vehicleCount}" placeholder="Make" required>

</div>

<div class="col-md-4 mb-2">

<input type="text" class="form-control" name="vehicleModel${vehicleCount}" placeholder="Model" required>

</div>

`;

document.getElementById('vehicleFields').appendChild(vehicleDiv);

});

 

// Form submission logic, local storage, and notification

document.getElementById('insuranceForm').addEventListener('submit', function(e) {

e.preventDefault();

const form = e.target;

const data = {

insuranceType: Array.from(form.querySelectorAll('input[name="insuranceType"]:checked')).map(i => i.value).join(', '),

zip: form.zip.value,

insuredLength: form.insuredLength.value,

provider: form.provider.value,

name: form.name.value,

address: form.address.value,

dob: form.dob.value,

email: form.email.value,

phone: form.phone.value,

contactPref: form.contactPref.value,

vehicles: []

};

for(let i=1; i<=vehicleCount; i++) {

data.vehicles.push({

year: form[`vehicleYear${i}`]?.value || "",

make: form[`vehicleMake${i}`]?.value || "",

model: form[`vehicleModel${i}`]?.value || ""

});

}

// Store data locally

localStorage.setItem('insuranceForm', JSON.stringify(data));

// Show notification

document.getElementById('formSuccess').classList.remove('d-none');

document.getElementById('userContact').textContent = data.contactPref.toLowerCase();

// Simulate sending email (for demo)

setTimeout(() => {

sendEmailToUser(data);

sendEmailToAdmin(data);

}, 1000);

// Reset form

form.reset();

document.getElementById('vehicleFields').innerHTML = '';

vehicleCount = 0;

});

}

}

// Restore form data if available

window.addEventListener('DOMContentLoaded', () => {

const saved = localStorage.getItem('insuranceForm');

if(saved) {

const data = JSON.parse(saved);

if(data.insuranceType) {

data.insuranceType.split(', ').forEach(type => {

const el = document.querySelector(`input[name="insuranceType"][value="${type}"]`);

if(el) el.checked = true;

});

}

if(data.zip) document.getElementById('zip').value = data.zip;

if(data.insuredLength) document.getElementById('insuredLength').value = data.insuredLength;

if(data.provider) document.getElementById('provider').value = data.provider;

if(data.name) document.getElementById('name').value = data.name;

if(data.address) document.getElementById('address').value = data.address;

if(data.dob) document.getElementById('dob').value = data.dob;

if(data.email) document.getElementById('email').value = data.email;

if(data.phone) document.getElementById('phone').value = data.phone;

if(data.contactPref) document.getElementById('contactPref').value = data.contactPref;

if(data.vehicles && data.vehicles.length) {

data.vehicles.forEach((v, idx) => {

document.getElementById('addVehicleBtn').click();

document.getElementsByName(`vehicleYear${idx+1}`)[0].value = v.year;

document.getElementsByName(`vehicleMake${idx+1}`)[0].value = v.make;

document.getElementsByName(`vehicleModel${idx+1}`)[0].value = v.model;

});

}

}

});
